31 std::vector<AddressMonitor::BalanceChange>
32 AddressMonitor::calculateBalanceChanges() {
33  auto monitored = monitoredAddresses();
34 
35  std::unordered_map<std::string, uint64_t> addressToIDs;
36  std::vector<std::string> addresses;
37  std::vector<BalanceChange> changes;
38 
39  // monitored -> map<address, id>
40  std::transform(
41  std::begin(monitored), std::end(monitored),
42  std::inserter(addressToIDs, addressToIDs.begin()), [](const auto& tup) {
43  return std::pair{std::move(std::get<1>(tup)), std::get<0>(tup)};
44  });
45 
46  // map<address, id> -> list<address>
47  std::transform(std::begin(addressToIDs), std::end(addressToIDs),
48  std::back_inserter(addresses),
49  [](const auto& pair) { return pair.first; });
50 
51  removeUnmonitoredAddresses(addressToIDs);
52  if (addresses.empty()) {
53  return {};
54  }
55  // TODO(th0br0) Figure out better failure pattern.
56  // At the moment, on failure, getBalances will return an empty
57  // list. Therefore, nothing will happen.
58  auto balances = _api->getBalances(std::move(addresses));
59 
60  if (!balances) {
61  return {};
62  }
63 
64  for (auto& pair : *balances) {
65  auto id = addressToIDs[pair.first];
66 
67  // This will create current (= 0) if the element does not exist.
68  int64_t prev = _balances[id];
69  int64_t delta = ((int64_t)pair.second) - prev;
70 
71  if (delta) {
72  changes.push_back({id, std::move(pair.first), pair.second, delta});
73  }
74  }
75 
76  return changes;
77 }

98 bool AddressMonitor::doTick() {
99  auto changes = calculateBalanceChanges();
100 
101  if (!changes.empty()) {
102  uint32_t currNumChanges = 0;
103  while (currNumChanges < changes.size()) {
104  auto currChanges =
105  nextBatch(changes, currNumChanges, FLAGS_balanceChangeBatchSize);
106  }
107  if (onBalancesChanged(changes)) {
108  persistBalanceChanges(std::move(changes));
109  }
110  }
111 
112  return true;

115 void AddressMonitor::removeUnmonitoredAddresses(
116  const std::unordered_map<std::string, uint64_t>& addressToIds) {
117  std::set<uint64_t> monitoredIds;
118  boost::copy(addressToIds | boost::adaptors::map_values,
119  std::inserter(monitoredIds, monitoredIds.begin()));
120 
121  std::set<uint64_t> balancesIds;
122  boost::copy(_balances | boost::adaptors::map_keys,
123  std::inserter(balancesIds, balancesIds.begin()));
124 
125  std::vector<uint64_t> unmonitoredBalances;
126  std::set_difference(balancesIds.begin(), balancesIds.end(),
127  monitoredIds.begin(), monitoredIds.end(),
128  std::back_inserter(unmonitoredBalances));
129 
130  for (auto id : unmonitoredBalances) {
131  _balances.erase(id);
132  }
133 }

135 template <typename T>
136 std::vector<T> AddressMonitor::nextBatch(const std::vector<T>& vec,
137  uint32_t& numBatchedEntries,
138  uint32_t batchSize) {
139  auto numToQuery = (vec.size() - numBatchedEntries) > batchSize
140  ? batchSize
141  : (vec.size() - numBatchedEntries);
142  auto startEntry = vec.begin() + numBatchedEntries;
143  auto currVec = std::vector<T>(startEntry, startEntry + numToQuery);
144 
145  numBatchedEntries += currVec.size();
146 
147  return currVec;
148 }
