66 void HubServer::initialise() {
67  if (FLAGS_signingMode == "remote") {
68  common::crypto::CryptoManager::get().setProvider(
69  std::make_unique<crypto::RemoteSigningProvider>(
70  FLAGS_signingProviderAddress, FLAGS_signingServerSslCert,
71  FLAGS_signingServerChainCert, FLAGS_signingServerKeyCert));
72  } else if (FLAGS_signingMode == "local") {
73  if (common::flags::FLAGS_salt.size() <= 20) {
74  LOG(FATAL) << "Salt must be more than 20 characters long.";
75  }
76  common::crypto::CryptoManager::get().setProvider(
77  std::make_unique<hub::crypto::LocalSigningProvider>(
78  common::flags::FLAGS_salt));
79  } else {
80  LOG(FATAL) << "Signing mode: \"" << FLAGS_signingMode
81  << "\" not recognized";
82  }
83 
84  initialiseAuthProvider();
85 
86  db::DBManager::get().loadConnectionConfigFromArgs();
87 
88  if (!authenticateSalt()) {
89  if (FLAGS_signingMode == "remote") {
90  LOG(FATAL) << "The remote crypto server could not be authenticated";
91  } else {
92  LOG(FATAL)
93  << "The provided salt or provider parameters are not valid for "
94  "this database. Did you mistype?";
95  }
96  }
97 
98  {
99  size_t portIdx = FLAGS_apiAddress.find(':');
100  auto host = FLAGS_apiAddress.substr(0, portIdx);
101  auto port = std::stoi(FLAGS_apiAddress.substr(portIdx + 1));
102 
103  _api = std::make_shared<cppclient::BeastIotaAPI>(host, port);
104  }
105 
106  iota::POWManager::get().setProvider(std::make_unique<iota::RemotePOW>(
107  _api, FLAGS_depth, FLAGS_minWeightMagnitude));
108 
109  _userAddressMonitor = std::make_unique<service::UserAddressMonitor>(
110  _api, std::chrono::milliseconds(FLAGS_monitorInterval));
111 
112  _attachmentService = std::make_unique<service::AttachmentService>(
113  _api, std::chrono::milliseconds(FLAGS_attachmentInterval));
114  _sweepService = std::make_unique<service::SweepService>(
115  _api, std::chrono::milliseconds(FLAGS_sweepInterval));
116 
117  ServerBuilder builder;
118 
119  builder.AddListeningPort(
120  common::flags::FLAGS_listenAddress,
121  makeCredentials(common::flags::FLAGS_authMode,
122  common::flags::FLAGS_sslCert, common::flags::FLAGS_sslKey,
123  common::flags::FLAGS_sslCA));
124  builder.RegisterService(&_service);
125 
126  _server = builder.BuildAndStart();
127  _userAddressMonitor->start();
128  _attachmentService->start();
129  _sweepService->start();
130 
131  LOG(INFO) << "Server listening on " << common::flags::FLAGS_listenAddress;
132 }

134 bool HubServer::authenticateSalt() const {
135  auto& connection = db::DBManager::get().connection();
136  auto addAndUuidRes = connection.selectFirstUserAddress();
137 
138  if (!addAndUuidRes.has_value()) return true;
139 
140  const auto& provider = common::crypto::CryptoManager::get().provider();
141  const auto& [existantAddress, uuid] = addAndUuidRes.value();
142 
143  auto maybeAddress = provider.getAddressForUUID(common::crypto::UUID(uuid));
144  if (!maybeAddress.has_value()) {
145  LOG(ERROR) << "Could not get address from provider.";
146  return false;
147  }
148  auto address = maybeAddress.value();
149 
150  return address.str_view() == existantAddress;
151 }

162  }
