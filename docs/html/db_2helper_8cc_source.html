206 std::vector<Sweep> helper<C>::getUnconfirmedSweeps(
207  C& connection, const std::chrono::system_clock::time_point& olderThan) {
208  db::sql::Sweep swp;
209  db::sql::SweepTails tls;
210 
211  std::vector<Sweep> sweeps;
212 
213  auto result =
214  connection(select(swp.id, swp.bundleHash, swp.trytes, swp.intoHubAddress)
215  .from(swp)
216  .where(swp.createdAt <= olderThan &&
217  !(exists(select(tls.hash).from(tls).where(
218  tls.sweep == swp.id && tls.confirmed == 1)))));
219 
220  constexpr size_t trytesPerTX = 2673;
221 
222  for (const auto& row : result) {
223  // Need to chunk this up.
224  std::vector<std::string> trytes;
225  auto combinedTrytes = row.trytes.value();
226  auto txCount = combinedTrytes.size() / trytesPerTX;
227 
228  for (size_t i = 0; i < txCount; i++) {
229  trytes.push_back(combinedTrytes.substr(i * trytesPerTX, trytesPerTX));
230  }
231 
232  sweeps.push_back({static_cast<uint64_t>(row.id), std::move(row.bundleHash),
233  std::move(trytes),
234  static_cast<uint64_t>(row.intoHubAddress)});
235  }
236 
237  return sweeps;
238 }

300 template <typename C>
301 std::vector<UserAccountBalanceEvent>
302 helper<C>::getAllUsersAccountBalancesSinceTimePoint(
303  C& connection, std::chrono::system_clock::time_point newerThan) {
304  db::sql::UserAccountBalance bal;
305  db::sql::UserAccount acc;
306  db::sql::Sweep swp;
307  db::sql::Withdrawal withdrawal;
308 
309  auto result = connection(
310  select(acc.identifier, bal.amount, bal.reason, bal.occuredAt,
311  withdrawal.uuid, swp.bundleHash)
312  .from(bal.join(acc)
313  .on(bal.userId == acc.id)
314  .left_outer_join(swp)
315  .on(bal.sweep == swp.id)
316  .left_outer_join(withdrawal)
317  .on(bal.withdrawal == withdrawal.id))
318  .where(bal.occuredAt >= newerThan &&
319  (bal.reason ==
320  static_cast<int>(UserAccountBalanceReason::SWEEP) ||
321  (bal.reason ==
322  static_cast<int>(UserAccountBalanceReason::WITHDRAWAL) ||
323  bal.reason ==
324  static_cast<int>(
325  UserAccountBalanceReason::WITHDRAWAL_CANCEL))))
326  .order_by(bal.occuredAt.asc()));
327 
328  std::vector<UserAccountBalanceEvent> balances;
329 
330  for (auto& row : result) {
331  std::chrono::time_point<std::chrono::system_clock> ts =
332  row.occuredAt.value();
333  auto reason = static_cast<UserAccountBalanceReason>(row.reason.value());
334  if (reason == UserAccountBalanceReason::SWEEP) {
335  balances.emplace_back(
336  UserAccountBalanceEvent{std::move(row.identifier), ts, row.amount,
337  reason, std::move(row.bundleHash), ""});
338  } else {
339  balances.emplace_back(UserAccountBalanceEvent{std::move(row.identifier),
340  ts, row.amount, reason, "",
341  std::move(row.uuid)});
342  }
343  }
344 
345  return balances;
346 }

424 template <typename C>
425 std::vector<SweepEvent> helper<C>::getSweeps(
426  C& connection, std::chrono::system_clock::time_point newerThan) {
427  db::sql::Sweep swp;
428  db::sql::Withdrawal withdrawal;
429  auto result = connection(
430  select(swp.id, swp.bundleHash, swp.createdAt, withdrawal.uuid)
431  .from(swp.left_outer_join(withdrawal).on(swp.id == withdrawal.sweep))
432  .where(swp.createdAt >= newerThan)
433  .order_by(swp.id.asc()));
434 
435  std::unordered_map<std::string, std::vector<std::string>> hashToUUIDs;
436 
437  int64_t id = -1;
438  std::vector<SweepEvent> events;
439  SweepEvent* currEvent;
440 
441  for (auto& row : result) {
442  if (row.id != id) {
443  id = row.id;
444  std::vector<std::string> uuids;
445  std::chrono::time_point<std::chrono::system_clock> ts =
446  row.createdAt.value();
447  currEvent = &events.emplace_back(
448  SweepEvent{std::move(row.bundleHash), ts, std::move(uuids)});
449  }
450 
451  if (!std::string{row.uuid}.empty()) {
452  auto& uuids = currEvent->withdrawalUUIDs;
453  uuids.emplace_back(std::string{std::move(row.uuid)});
454  }
455  }
456 
457  return events;
458 }

659 template <typename C>
660 std::vector<TransferInput> helper<C>::getHubInputsForSweep(
661  C& connection, uint64_t requiredAmount,
662  const std::chrono::system_clock::time_point& olderThan) {
663  db::sql::HubAddress add;
664  db::sql::HubAddressBalance bal;
665  db::sql::Sweep swp;
666 
667  // 1. Get all available unused Hub addresses
668  auto availableAddressesResult = connection(
669  select(add.id, add.address, add.balance, add.seedUuid)
670  .from(add)
671  .where(add.createdAt < olderThan && add.isColdStorage == 0 &&
672  add.balance > 0 &&
673  !exists(select(bal.id).from(bal).where(
674  bal.hubAddress == add.id &&
675  bal.reason == static_cast<int>(
676  HubAddressBalanceReason::OUTBOUND)))));
677 
678  std::vector<TransferInput> availableInputs;
679  std::vector<int64_t> addressIds;
680  for (const auto& row : availableAddressesResult) {
681  auto id = row.id;
682 
683  TransferInput input = {id, 0, common::crypto::Address(row.address.value()),
684  common::crypto::UUID(row.seedUuid.value()),
685  static_cast<uint64_t>(row.balance)};
686 
687  addressIds.push_back(id);
688  availableInputs.emplace_back(std::move(input));
689  }
690 
691  // 2. Only select those with all INBOUND sweeps confirmed
692  auto confirmedAddresses = connection(
693  select(swp.intoHubAddress)
694  .where(swp.intoHubAddress.in(sqlpp::value_list(addressIds)))
695  .from(swp)
696  .group_by(swp.intoHubAddress)
697  .having(sum(swp.confirmed) == count(swp.id)));
698 
699  std::unordered_set<int64_t> confirmedAddressIds;
700 
701  for (const auto& row : confirmedAddresses) {
702  confirmedAddressIds.insert(row.intoHubAddress);
703  }
704 
705  availableInputs.erase(
706  std::remove_if(availableInputs.begin(), availableInputs.end(),
707  [&confirmedAddressIds](const auto& input) {
708  return confirmedAddressIds.find(input.addressId) ==
709  confirmedAddressIds.end();
710  }),
711  availableInputs.end());
712 
713  // Sort lowest to highest
714  std::sort(availableInputs.begin(), availableInputs.end(),
715  [](const auto& a, const auto& b) { return a.amount < b.amount; });
716 
717  std::vector<TransferInput> selectedInputs;
718  auto it = availableInputs.begin();
719 
720  uint64_t total = 0;
721  while (it != std::end(availableInputs) && total < requiredAmount) {
722  total += it->amount;
723  selectedInputs.emplace_back(std::move(*it));
724  it++;
725  }
726 
727  return selectedInputs;
728 }
