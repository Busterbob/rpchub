46 std::string_view HMACProvider::stripEncodedPrefix(const std::string_view& e) {
47  // strip prefix from encoded
48  // Encoded example with prefix:
49  // $argon2id$v=19$m=131072,t=4,p=1$c29tZXNhbHQ$xLJXnoh1+mb/bqTS72pL+pBbH1+sih7KFiZV1k4fdmM
50  // After stripping: c29tZXNhbHQ$xLJXnoh1+mb/bqTS72pL+pBbH1+sih7KFiZV1k4fdmM
51 
52  static constexpr uint8_t NUM_DOLLAR_SIGNS_IN_PREFIX = 4;
53 
54  auto prefixSize = 0;
55  uint8_t dollarSignCounter = 0;
56  std::string_view curr(e);
57  while (dollarSignCounter < NUM_DOLLAR_SIGNS_IN_PREFIX) {
58  curr = e.substr(prefixSize, KEY_SIZE);
59  auto pos = curr.find_first_of('$');
60  if (pos == std::string::npos) {
61  LOG(ERROR) << "Failed in " << __FUNCTION__;
62  break;
63  }
64  curr = curr.substr(0, pos + 1);
65  prefixSize += curr.size();
66  dollarSignCounter++;
67  }
68 
69  return e.substr(prefixSize, KEY_SIZE);
70 }
