51 bool UserAddressMonitor::onBalancesChanged(
52  const std::vector<AddressMonitor::BalanceChange>& changes) {
53  LOG(INFO) << "Processing changes. Total: " << changes.size();
54  auto& connection = db::DBManager::get().connection();
55  bool error = true;
56 
57  auto transaction = connection.transaction();
58 
59  try {
60  std::vector<std::string> addresses;
61  std::transform(std::begin(changes), std::end(changes),
62  std::back_inserter(addresses),
63  [](const auto& change) { return change.address; });
64 
65  std::vector<uint64_t> ids;
66  std::transform(std::begin(changes), std::end(changes),
67  std::back_inserter(ids),
68  [](const auto& change) { return change.addressId; });
69 
70  auto confirmedBundlesMap = _api->getConfirmedBundlesForAddresses(addresses);
71 
72  auto tailsToAddresses = connection.tailsForUserAddresses(ids);
73 
74  for (const auto& change : changes) {
75  std::set<std::string> tails;
76  auto tailsER = tailsToAddresses.equal_range(change.addressId);
77  for (auto it = tailsER.first; it != tailsER.second; ++it) {
78  tails.insert(std::move(it->second));
79  }
80 
81  auto confirmedBundlesER = confirmedBundlesMap.equal_range(change.address);
82  std::vector<cppclient::Bundle> unknownBundles;
83 
84  for (auto it = confirmedBundlesER.first; it != confirmedBundlesER.second;
85  ++it) {
86  auto& ctail = it->second[0].hash;
87  if (tails.find(ctail) == tails.end()) {
88  unknownBundles.push_back(std::move(it->second));
89  }
90  }
91 
92  int64_t aggregateSum = 0;
93 
94  for (const auto& bundle : unknownBundles) {
95  auto& tail = bundle[0].hash;
96 
97  for (const auto& tx : bundle) {
98  if (tx.address == change.address && tx.value) {
99  aggregateSum += tx.value;
100 
101  if (tx.value < 0) {
102  LOG(FATAL) << "Funds have been spent from deposit address: "
103  << tx.address << " (bundle hash: " << tx.bundleHash
104  << ")";
105  } else {
106  LOG(INFO) << "Creating UserAddressBalance(" << change.addressId
107  << "," << tx.value << ",DEPOSIT," << tail << ")";
108 
109  connection.createUserAddressBalanceEntry(
110  change.addressId, tx.value,
111  db::UserAddressBalanceReason::DEPOSIT, tail, {});
112  }
113  }
114  }
115  }
116 
117  if (aggregateSum != change.delta) {
118  LOG(ERROR) << "On-Tangle sum for address: " << change.address
119  << " did not match expected. Expected (getBalances): "
120  << change.delta << " observed: " << aggregateSum;
121  goto cleanup;
122  }
123 
124  if (!validateBalanceIsConsistent(change.address, change.addressId)) {
125  goto cleanup;
126  }
127  }
128 
129  error = false;
130  cleanup:
131  if (error) {
132  LOG(ERROR) << "Controlled rollback.";
133 
134  transaction->rollback();
135  } else {
136  transaction->commit();
137  return true;
138  }
139  } catch (sqlpp::exception& ex) {
140  LOG(ERROR) << " Commit failed: " << ex.what();
141 
142  try {
143  transaction->rollback();
144  } catch (const std::exception& ex) {
145  LOG(ERROR) << "Rollback failed: " << ex.what();
146  }
147  }
148 
149  return false;
150 }

152 bool UserAddressMonitor::validateBalanceIsConsistent(const std::string& address,
153  uint64_t addressId) {
154  auto& connection = db::DBManager::get().connection();
155  const auto& iriBalances = _api->getBalances({address});
156  const auto& dbBalances = connection.getTotalAmountForAddresses({addressId});
157 
158  if (!iriBalances) {
159  LOG(ERROR) << "IRI getBalances call failed.";
160  return false;
161  }
162 
163  if (iriBalances->size() != 1 || dbBalances.size() != 1) {
164  LOG(ERROR) << "Could not get balance for address: \n" + address;
165  return false;
166  }
167 
168  uint64_t observedBalance = iriBalances->begin()->second;
169  uint64_t expectedBalance = dbBalances.begin()->second;
170  if (observedBalance < expectedBalance) {
171  LOG(ERROR) << "Observed sum is less than expected. Expected (db): "
172  << expectedBalance
173  << " Observed (getBalances): " << observedBalance;
174  return false;
175  }
176 
177  return true;
178 }
