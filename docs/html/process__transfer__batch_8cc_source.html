41  const auto& t = request->transfers(i);

53  if (!validationStatus.ok()) {

59  const hub::rpc::ProcessTransferBatchRequest_Transfer& t =
60  request->transfers(i);

85 grpc::Status ProcessTransferBatch::validateTransfers(
86  const hub::rpc::ProcessTransferBatchRequest* request,
87  const std::map<std::string, int64_t>& identifierToId) noexcept {
88  auto& connection = db::DBManager::get().connection();
89  std::unordered_map<std::string, int64_t> userToTransferAmount;
90 
91  uint64_t totalBatchSum = 0;
92 
93  bool zeroAmount = false;
94  // validate amounts are not zero
95  for (auto i = 0; i < request->transfers_size(); ++i) {
96  const auto& t = request->transfers(i);
97  if (t.amount() == 0) {
98  zeroAmount = true;
99  break;
100  }
101  totalBatchSum += t.amount();
102  if (userToTransferAmount.find(t.userid()) == userToTransferAmount.end()) {
103  userToTransferAmount[t.userid()] = 0;
104  }
105  userToTransferAmount[t.userid()] += t.amount();
106  }
107 
108  if (zeroAmount) {
109  return grpc::Status(grpc::StatusCode::FAILED_PRECONDITION,
110  errorToString(hub::rpc::ErrorCode::BATCH_INVALID));
111  }
112  if (totalBatchSum != 0) {
113  return grpc::Status(grpc::StatusCode::FAILED_PRECONDITION,
114  errorToString(hub::rpc::ErrorCode::BATCH_AMOUNT_ZERO));
115  }
116 
117  std::set<uint64_t> userIds;
118  boost::copy(identifierToId | boost::adaptors::map_values,
119  std::inserter(userIds, userIds.begin()));
120  auto userToBalance = connection.getTotalAmountForUsers(userIds);
121  for (auto& kv : userToTransferAmount) {
122  uint64_t currId = identifierToId.at(kv.first);
123 
124  auto availableBalance = userToBalance.at(currId);
125  auto balanceChange = kv.second;
126 
127  if ((availableBalance + balanceChange) < 0) {
128  return grpc::Status(
129  grpc::StatusCode::FAILED_PRECONDITION,
130  errorToString(hub::rpc::ErrorCode::BATCH_INCONSISTENT));
131  }
132  }
133 
134  return grpc::Status::OK;
135 }
