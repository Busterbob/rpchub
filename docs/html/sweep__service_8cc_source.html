62 std::tuple<common::crypto::Hash, std::string> SweepService::createBundle(
63  const std::vector<db::TransferInput>& deposits,
64  const std::vector<db::TransferInput>& hubInputs,
65  const std::vector<db::TransferOutput>& withdrawals,
66  const db::TransferOutput& hubOutput) {
67  auto& dbConnection = db::DBManager::get().connection();
68  auto& cryptoProvider = common::crypto::CryptoManager::get().provider();
69 
70  // 5.1. Generate bundle hash & transactions
71  IOTA::Models::Bundle bundle;
72  {
73  // timestamp.
74  auto timestamp = std::chrono::duration_cast<std::chrono::seconds>(
75  std::chrono::system_clock::now().time_since_epoch())
76  .count();
77 
78  // inputs: deposits
79  for (const auto& deposit : deposits) {
80  IOTA::Models::Transaction tx;
81  tx.setAddress(deposit.address.str());
82  tx.setTimestamp(timestamp);
83  tx.setValue((-1uLL) * deposit.amount);
84 
85  bundle.addTransaction(tx,
86  cryptoProvider.securityLevel(deposit.uuid).value());
87  }
88  // inputs: hubInputs
89  for (const auto& input : hubInputs) {
90  IOTA::Models::Transaction tx;
91  tx.setAddress(input.address.str());
92  tx.setTimestamp(timestamp);
93  tx.setValue((-1uLL) * input.amount);
94 
95  bundle.addTransaction(tx,
96  cryptoProvider.securityLevel(input.uuid).value());
97  }
98  // outputs: withdrawals
99  for (const auto& wd : withdrawals) {
100  IOTA::Models::Transaction tx;
101  tx.setAddress(wd.payoutAddress.str());
102  tx.setTimestamp(timestamp);
103  tx.setValue(wd.amount);
104 
105  if (wd.tag) {
106  tx.setObsoleteTag(wd.tag->str());
107  }
108 
109  bundle.addTransaction(tx, 1);
110  }
111 
112  // output: hubOutput
113  IOTA::Models::Transaction tx;
114  tx.setAddress(hubOutput.payoutAddress.str());
115  tx.setTimestamp(timestamp);
116  tx.setValue(hubOutput.amount);
117 
118  bundle.addTransaction(tx, 1);
119  }
120 
121  bundle.finalize();
122 
123  common::crypto::Hash bundleHash(bundle.getHash());
124 
125  // 5.2 Generate signatures
126  std::unordered_map<common::crypto::Address, std::string> signaturesForAddress;
127 
128  for (const auto& in : deposits) {
129  signaturesForAddress[in.address] =
130  cryptoProvider.getSignatureForUUID(in.uuid, bundleHash).value();
131  }
132  for (const auto& in : hubInputs) {
133  signaturesForAddress[in.address] =
134  cryptoProvider.getSignatureForUUID(in.uuid, bundleHash).value();
135  }
136 
137  auto it = bundle.getTransactions().begin();
138 
139  while (it != bundle.getTransactions().end()) {
140  auto& tx = *it;
141 
142  if (tx.getValue() >= 0) {
143  tx.setSignatureFragments(EMPTY_FRAG);
144  tx.setNonce(EMPTY_NONCE);
145  tx.setTrunkTransaction(EMPTY_HASH);
146  tx.setBranchTransaction(EMPTY_HASH);
147 
148  if (tx.getObsoleteTag().empty()) {
149  tx.setObsoleteTag(EMPTY_NONCE);
150  }
151 
152  ++it;
153  continue;
154  }
155 
156  std::string_view signature = signaturesForAddress.at(
157  common::crypto::Address(tx.getAddress().toTrytes()));
158 
159  while (!signature.empty()) {
160  (*it).setSignatureFragments(
161  std::string(signature.substr(0, FRAGMENT_LEN)));
162  (*it).setNonce(EMPTY_NONCE);
163  (*it).setTrunkTransaction(EMPTY_HASH);
164  (*it).setBranchTransaction(EMPTY_HASH);
165 
166  signature.remove_prefix(FRAGMENT_LEN);
167 
168  ++it;
169  }
170  }
171 
172  std::ostringstream bundleTrytesOS;
173  for (const auto& tx : bundle.getTransactions()) {
174  bundleTrytesOS << tx.toTrytes();
175  }
176 
177  return {std::move(bundleHash), bundleTrytesOS.str()};
178 }

180 void SweepService::persistToDatabase(
181  std::tuple<common::crypto::Hash, std::string> bundle,
182  const std::vector<db::TransferInput>& deposits,
183  const std::vector<db::TransferInput>& hubInputs,
184  const std::vector<db::TransferOutput>& withdrawals,
185  const db::TransferOutput& hubOutput) {
186  auto& dbConnection = db::DBManager::get().connection();
187 
188  // 6.1. Insert sweep.
189  auto sweepId = dbConnection.createSweep(std::get<0>(bundle),
190  std::get<1>(bundle), hubOutput.id);
191 
192  // 6.2. Change Hub address balances
193 
194  dbConnection.createHubAddressBalanceEntry(
195  hubOutput.id, hubOutput.amount, db::HubAddressBalanceReason::INBOUND,
196  sweepId);
197  for (const auto& input : hubInputs) {
198  dbConnection.createHubAddressBalanceEntry(
199  input.addressId, -input.amount, db::HubAddressBalanceReason::OUTBOUND,
200  sweepId);
201  }
202 
203  // 6.3. Update withdrawal sweep id
204  for (const auto& withdrawal : withdrawals) {
205  dbConnection.setWithdrawalSweep(withdrawal.id, sweepId);
206  }
207 
208  // 6.4. Change User address balances
209  for (const auto& input : deposits) {
210  dbConnection.createUserAddressBalanceEntry(
211  input.addressId, -input.amount, db::UserAddressBalanceReason::SWEEP, {},
212  sweepId);
213 
214  dbConnection.createUserAccountBalanceEntry(
215  input.userId, input.amount, db::UserAccountBalanceReason::SWEEP,
216  sweepId);
217  }
218 }

220 bool SweepService::doTick() {
221  LOG(INFO) << "Starting sweep.";
222  auto sweepStart = std::chrono::system_clock::now();
223 
224  auto& dbConnection = db::DBManager::get().connection();
225  auto transaction = dbConnection.transaction();
226 
227  try {
228  if (!_api->isNodeSolid()) {
229  LOG(INFO) << "Node is not solid. Aborting sweep.";
230 
231  transaction->rollback();
232  return true;
233  }
234 
235  // 1. Get up to X withdrawal requests
236  auto withdrawals = dbConnection.getWithdrawalsForSweep(
237  FLAGS_sweep_max_withdraw, sweepStart);
238 
239  auto requiredOutput =
240  std::accumulate(withdrawals.cbegin(), withdrawals.cend(), 0uLL,
241  [](uint64_t a, const auto& b) { return a + b.amount; });
242 
243  LOG(INFO) << "Found " << withdrawals.size()
244  << " withdrawals with total amount: " << requiredOutput;
245 
246  // 2. Get up to Y sweep inputs (= deposits)
247  auto deposits =
248  dbConnection.getDepositsForSweep(FLAGS_sweep_max_deposit, sweepStart);
249  auto depositsTotal =
250  std::accumulate(deposits.cbegin(), deposits.cend(), 0uLL,
251  [](uint64_t a, const auto& b) { return a + b.amount; });
252 
253  LOG(INFO) << "Found " << deposits.size()
254  << " deposits with total amount: " << depositsTotal;
255 
256  // 2.1. Abort if no deposits and no withdrawals were found.
257  if (deposits.size() == 0 && withdrawals.size() == 0) {
258  LOG(INFO) << "No deposits or withdrawal requests found. Aborting sweep.";
259 
260  transaction->rollback();
261  return true;
262  }
263 
264  // 3. Find hub addresses to fill missing input balance.
265  std::vector<db::TransferInput> hubInputs;
266  uint64_t hubInputTotal = 0;
267 
268  if (depositsTotal < requiredOutput) {
269  auto missing = requiredOutput - depositsTotal;
270 
271  LOG(INFO) << "Need to fill " << missing << " via Hub addresses.";
272 
273  hubInputs = dbConnection.getHubInputsForSweep(missing, sweepStart);
274 
275  hubInputTotal = std::accumulate(
276  hubInputs.cbegin(), hubInputs.cend(), 0uLL,
277  [](uint64_t a, const auto& b) { return a + b.amount; });
278 
279  if (hubInputTotal < missing) {
280  LOG(ERROR)
281  << "Not enough balance available in hub addresses right now. "
282  << "Needed: " << missing << " but only found: " << hubInputTotal
283  << ". Ignoring withdrawals.";
284 
285  hubInputTotal = 0;
286  hubInputs.clear();
287  requiredOutput = 0;
288  withdrawals.clear();
289  }
290  }
291 
292  // 3.1. Abort if no deposits or withdrawals found.
293  if (deposits.size() == 0 && withdrawals.size() == 0) {
294  LOG(INFO) << "No deposits or withdrawal requests found. Aborting sweep.";
295 
296  transaction->rollback();
297  return true;
298  }
299 
300  LOG(INFO) << "Found " << hubInputs.size()
301  << " hub inputs to fill missing funds.";
302 
303  // 4. Determine Hub output address
304  auto remainder = (hubInputTotal + depositsTotal) - requiredOutput;
305  auto hubOutput = getHubOutput(remainder);
306 
307  LOG(INFO) << "Will move " << remainder
308  << " into new Hub address: " << hubOutput.payoutAddress.str();
309  // 5. Generate bundle
310  auto bundle = createBundle(deposits, hubInputs, withdrawals, hubOutput);
311 
312  // 6. Commit to DB
313  persistToDatabase(bundle, deposits, hubInputs, withdrawals, hubOutput);
314 
315  transaction->commit();
316  LOG(INFO) << "Sweep complete.";
317  } catch (const std::exception& ex) {
318  try {
319  transaction->rollback();
320  } catch (const std::exception& ex) {
321  LOG(ERROR) << "Sweep. Rollback failed: " << ex.what();
322  }
323 
324  LOG(ERROR) << "Sweep failed: " << ex.what();
325  }
326 
327  // PoW is handled by attachment service
328 
329  return true;
330 }
