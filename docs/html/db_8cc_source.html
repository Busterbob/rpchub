55 void DBManager::executeFile(const std::string& fname) {
56  auto& conn = connection();
57 
58  std::ifstream fSchema(fname);
59  std::string schema((std::istreambuf_iterator<char>(fSchema)),
60  std::istreambuf_iterator<char>());
61  std::stringstream ss(schema);
62 
63  std::string line;
64  std::string cmd;
65 
66  while (std::getline(ss, line, '\n')) {
67  auto pos = line.find_first_not_of(' ');
68  if (pos != std::string::npos && line[pos] == '-') {
69  continue;
70  }
71 
72  cmd += line;
73  cmd += " ";
74 
75  if (line.find(';') != std::string::npos) {
76  conn.execute(cmd);
77  cmd = "";
78  }
79  }
80 }
