37 bool AttachmentService::checkSweepTailsForConfirmation(
38  db::Connection& connection, const db::Sweep& sweep,
39  const std::vector<std::string>& tails) {
40  auto confirmedTails = _api->filterConfirmedTails(tails, {});
41 
42  LOG(INFO) << "Sweep: " << sweep.id << " (" << sweep.bundleHash
43  << ") has: " << confirmedTails.size() << " confirmed tails.";
44 
45  if (confirmedTails.size() == 1) {
46  const auto& tail = *confirmedTails.cbegin();
47  LOG(INFO) << "Marking tail as confirmed: " << tail;
48  connection.markTailAsConfirmed(tail);
49  return true;
50  } else if (confirmedTails.size() > 1) {
51  LOG(FATAL) << "More than one confirmed tail!!!";
52  }
53 
54  return false;
55 }

57 bool AttachmentService::checkForUserReattachment(
58  db::Connection& connection, const db::Sweep& sweep,
59  const std::vector<std::string>& knownTails) {
60  auto bundleTransactionHashes = _api->findTransactions(
61  {}, std::vector<std::string>{sweep.bundleHash}, {});
62  auto bundleTransactions = _api->getTransactions(bundleTransactionHashes);
63 
64  // Remove non-tails or tails that we know of
65  bundleTransactions.erase(
66  std::remove_if(bundleTransactions.begin(), bundleTransactions.end(),
67  [&knownTails](const auto& tx) {
68  return
69  // Not a tail
70  tx.currentIndex != 0 ||
71  // Already know about this one
72  std::find(knownTails.begin(), knownTails.end(),
73  tx.hash) != knownTails.end();
74  }),
75  bundleTransactions.end());
76 
77  if (bundleTransactions.size() != 0) {
78  // Third-party (i.e. user erattachments!)
79  std::vector<std::string> userTails;
80 
81  for (const auto& ref : bundleTransactions) {
82  userTails.push_back(ref.hash);
83  }
84 
85  auto confirmedTails = _api->filterConfirmedTails(userTails, {});
86  if (confirmedTails.size() == 1) {
87  const auto& tail = *confirmedTails.cbegin();
88  LOG(INFO) << "Inserting confirmed user-attached tail: " << tail;
89  connection.createTail(sweep.id, tail);
90 
91  LOG(INFO) << "Marking tail as confirmed: " << tail;
92  connection.markTailAsConfirmed(tail);
93  return true;
94  }
95 
96  if (confirmedTails.size() > 1) {
97  // This should be impossible since bundle spends all funds
98  // in any input address, meaning that it won't be validated after one
99  // bundle has been confirmed
100  LOG(FATAL) << "Bundle hash: " << sweep.bundleHash
101  << " has more than one confirmed tail";
102  }
103 
104  // This also makes sure that we only add bundles that IRI has seen
105  // fully.
106  auto consistentTails = _api->filterConsistentTails(userTails);
107 
108  if (consistentTails.size() != 0) {
109  for (const auto& tail : consistentTails) {
110  LOG(INFO) << "Inserting Unconfirmed user-attached tail: " << tail;
111  connection.createTail(sweep.id, tail);
112  }
113  }
114  }
115  return false;
116 }

118 void AttachmentService::reattachSweep(db::Connection& dbConnection,
119  const iota::POWProvider& powProvider,
120  const db::Sweep& sweep) {
121  auto attachedTrytes = powProvider.performPOW(sweep.trytes);
122  // Get tail hash of first tx:
123  auto tailHash = iota_digest(attachedTrytes[0].c_str());
124  LOG(INFO) << "Reattached sweep " << sweep.id << " as: " << tailHash;
125 
126  _api->storeTransactions(attachedTrytes);
127  _api->broadcastTransactions(attachedTrytes);
128 
129  dbConnection.createTail(sweep.id, tailHash);
130 
131  std::free(tailHash);
132 }

134 void AttachmentService::promoteSweep(db::Connection& connection,
135  const iota::POWProvider& powProvider,
136  const db::Sweep& sweep,
137  const common::crypto::Hash& tailHash) {
138  auto toApprove = _api->getTransactionsToApprove(0, {tailHash.str()});
139  auto timestamp = std::chrono::duration_cast<std::chrono::seconds>(
140  std::chrono::system_clock::now().time_since_epoch())
141  .count();
142 
143  IOTA::Models::Bundle bundle;
144  IOTA::Models::Transaction otx;
145 
146  otx.setAddress(
147  "SOME9RANDOM9HUB9BEING9REATTACHED9999999999999999999999999999999999999999"
148  "999999999");
149  otx.setValue(0);
150  otx.setTimestamp(timestamp);
151 
152  bundle.addTransaction(otx, 1);
153  bundle.addTrytes({""});
154  bundle.finalize();
155 
156  const auto& tx = bundle.getTransactions()[0];
157 
158  auto attachedTrytes = powProvider.doPOW(
159  {tx.toTrytes()}, toApprove.trunkTransaction, toApprove.branchTransaction);
160 
161  _api->storeTransactions(attachedTrytes);
162  _api->broadcastTransactions(attachedTrytes);
163 
164  auto promotionHash = iota_digest(attachedTrytes[0].c_str());
165  LOG(INFO) << "Issued promotion for sweep " << sweep.id << ": "
166  << promotionHash;
167  std::free(promotionHash);
168 }

170 bool AttachmentService::doTick() {
171  // We're splitting processing out to separate methods on purpose.
172  // This means, that we can't really batch API requests across sweeps.
173  // Shouldn't matter though as we simply assume that the IOTA node is only used
174  // by Hub.
175 
176  auto& connection = db::DBManager::get().connection();
177  auto& powProvider = iota::POWManager::get().provider();
178 
179  auto tickStart = std::chrono::system_clock::now();
180 
181  // 1. Get Unconfirmed sweeps from database.
182  auto unconfirmedSweeps = connection.getUnconfirmedSweeps(tickStart);
183  LOG(INFO) << "Found " << unconfirmedSweeps.size() << " unconfirmed sweeps.";
184 
185  for (const auto& sweep : unconfirmedSweeps) {
186  auto transaction = connection.transaction();
187  auto milestone = _api->getNodeInfo().value().latestMilestone;
188 
189  try {
190  // 2. Get (tails, timestamp) for these sweeps
191  const auto initialSweepInfos = connection.getTailsForSweep(sweep.id);
192 
193  std::vector<std::string> initialSweepTailHashes;
194  std::transform(std::begin(initialSweepInfos), std::end(initialSweepInfos),
195  std::back_inserter(initialSweepTailHashes),
196  [](const auto& sweepTail) { return sweepTail.hash; });
197 
198  // 3. Check if one of the tails is confirmed.
199  // 4. If not, check if a user reattachment was confirmed
200  if (checkSweepTailsForConfirmation(connection, sweep,
201  initialSweepTailHashes) ||
202  checkForUserReattachment(connection, sweep, initialSweepTailHashes)) {
203  LOG(INFO) << "Sweep " << sweep.id << " is CONFIRMED.";
204  } else {
205  LOG(INFO) << "Sweep " << sweep.id << " is still unconfirmed.";
206 
207  // 5. If not, check if at least one of the tails per sweep is still
208  // promotable
209  // Requerying list of tails because `checkForUserReattachment` might
210  // have added some.
211  const auto sweepTails = connection.getTailsForSweep(sweep.id);
212  std::vector<std::string> sweepTailHashes;
213  std::transform(std::begin(sweepTails), std::end(sweepTails),
214  std::back_inserter(sweepTailHashes),
215  [](const auto& sweepTail) { return sweepTail.hash; });
216  auto consistentTails = _api->filterConsistentTails(sweepTailHashes);
217  // 5.1. If yes, pick most recent and promote.
218  if (consistentTails.size() > 0) {
219  // Get newest tail.
220  auto toPromote = std::find_if(
221  sweepTails.begin(), sweepTails.end(),
222  [&consistentTails](const auto& tail) {
223  return consistentTails.find(tail.hash) !=
224  consistentTails.end() &&
225  std::chrono::duration_cast<std::chrono::minutes>(
226  std::chrono::system_clock::now() - tail.createdAt)
227  .count() < FLAGS_max_promotion_age;
228  });
229 
230  if (toPromote != sweepTails.end()) {
231  // Promotion can fail if getTransactionsToApprove fails!
232  // In this case, we just catch and reattach.
233  try {
234  promoteSweep(connection, powProvider, sweep,
235  common::crypto::Hash((*toPromote).hash));
236  } catch (const std::exception& ex) {
237  LOG(INFO) << "Promotion failed. Reattaching.";
238  reattachSweep(connection, powProvider, sweep);
239  }
240  } else {
241  reattachSweep(connection, powProvider, sweep);
242  }
243  } else {
244  // 6. If not, reattach and commit tail to DB.
245  reattachSweep(connection, powProvider, sweep);
246  }
247  }
248 
249  transaction->commit();
250  } catch (const std::exception& ex) {
251  LOG(ERROR) << "Sweep " << sweep.id
252  << " failed to commit to DB: " << ex.what();
253 
254  try {
255  transaction->rollback();
256  } catch (const std::exception& ex) {
257  LOG(ERROR) << "Sweep " << sweep.id << " rollback failed: " << ex.what();
258  }
259  }
260  }
261 
262  return true;
263 }
